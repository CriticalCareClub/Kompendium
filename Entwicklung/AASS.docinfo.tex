\def\VarDocVersionPrimaryExplicit			{{\VarDocVersionMajor}.{\VarDocVersionMinor}.{\VarDocVersionRevision}}



% \VarDocVersionPrimary: Kürzere Primary-Versionskennung wenn x.0.0 und x.y.0:
\ifthenelse{\equal{\VarDocVersionRevision}{0} \AND \equal{\VarDocVersionMinor}{0}}	{\def\VarDocVersionPrimary{\VarDocVersionMajor}}{
  % x.0.0 --> x
  % x.y.0 --> x.y
  \ifthenelse{\equal{\VarDocVersionRevision}{0}}									{\def\VarDocVersionPrimary{{\VarDocVersionMajor}.{\VarDocVersionMinor}}}{
  % x.y.z --> x-y-z
  \def\VarDocVersionPrimary															{\VarDocVersionPrimaryExplicit}}}


\ifthenelse{\equal{\VarDocVersionType}{nightly}}
  {
    \def\VarDocVersion						{{\VarDocVersionPrimary}.{\VarDocVersionType}.{\pdfdate}}
    \def\VarDocVersionFull					{{\VarDocVersionMajor}.{\VarDocVersionMinor}.{\VarDocVersionType}.{\pdfdate}.var-{\VarDocVariant}}
    \def\VarDocVersionExplicit				{{\VarDocVersionPrimaryExplicit}.{\VarDocVersionType}.{\pdfdate}.var-{\VarDocVariant}}
    \def\VarDocVersionTypeName				{Entwicklungsversion}
    \def\VarDocVersionTypeDescription		{Für den internen Gebrauch.}
  }
  {}

    \def\VarDocVersion						{{\VarDocVersionPrimary}.{\VarDocVersionType}.{\pdfdate}}

    \def\VarDocVersionExplicit				{{\VarDocVersionPrimaryExplicit}.{\VarDocVersionType}.{\pdfdate}.var-{\VarDocVariant}}
    \def\VarDocVersionTypeName				{Entwicklungszwischenversion}
    \def\VarDocVersionTypeDescription		{Für den internen Gebrauch.}

  {}

    \def\VarDocVersionTypeName				{Iterationsversion}
